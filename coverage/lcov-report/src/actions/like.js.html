import axios from 'axios';
import { LIKE, UNLIKE, LIKE_ERROR } from './types';
 
export const likeAction = (payload) = ({type: LIKE, payload});
export const unlikeAction = (payload) = ({type: UNLIKE, payload});
export const likeError = (error) = ({type: LIKE_ERROR, error});
 
export const like = () => dispatch => {
  return axios.post(`localhost:3000/api/articles`)
  .then((response) => {
    dispatch(likeAction(response));
  })
  .catch(({message}) => {
    dispatch(likeError(message));
  });
}
 
export const unlike = () => dispatch => {
  return axios.delete(`localhost:3000/api/articles`)
  .then((response) => {
    dispatch(unlikeAction(response));
  })
  .catch(({message}) => {
    dispatch(likeError(message));
  });
}
